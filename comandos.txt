create table PUBLICACION( id_publicacion integer generated by default as Identity primary key,nombre varchar(75),detalle_producto varchar(300),precio number(9,3),fotoproducto varchar (250),id_usuario integer,categoria varchar(50),status integer default 1,constraint user_publi_fk foreign key (id_usuario) references USUARIO(id_usuario),constraint catego_public_fk foreign key (categoria) references CATEGORIA(nombre));

create table PALABRASCLAVES(id_key integer generated by default as Identity primary key,palabra varchar(50),id_producto integer,constraint id_pro_keyfk foreign key (id_producto) references PUBLICACION(id_publicacion));

create table COMENTARIOS(id_comentario integer generated by default as Identity primary key,fecha date default sysdate not null,autor varchar(50),descripcion varchar(300),id_publicacion integer,constraint produ_comment_fk foreign key (id_publicacion) references PUBLICACION(id_publicacion));
